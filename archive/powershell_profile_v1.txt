function Barri-Scaffold {
    <# .SYNOPSIS Opens VS 2022 Dev Cmd context, navigates to Barri, and runs the scaffolder. #>
    $targetDir  = "C:\dev\BusinessSystems.Barri\Trayport.BS.Entities.Barri"
    $connString = "Data Source=localhost;Initial Catalog=TRAYINVOICE;Integrated Security=True;TrustServerCertificate=True;Encrypt=True"

    $progFiles = [System.Environment]::GetFolderPath([System.Environment+SpecialFolder]::ProgramFilesX86)
    if ([string]::IsNullOrEmpty($progFiles)) { $progFiles = [System.Environment]::GetFolderPath([System.Environment+SpecialFolder]::ProgramFiles) }
    
    $vsWherePath = Join-Path $progFiles "Microsoft Visual Studio\Installer\vswhere.exe"
    if (-not (Test-Path $vsWherePath)) { Write-Error "Could not find vswhere.exe"; return }

    $vsInstallPath = & $vsWherePath -latest -version "[17.0,18.0)" -products * -requires Microsoft.VisualStudio.Component.Roslyn.Compiler -property installationPath
    if (-not $vsInstallPath) { Write-Error "VS 2022 not found."; return }

    $vsDevCmd = "$vsInstallPath\Common7\Tools\VsDevCmd.bat"
    $innerCommand = "`"$vsDevCmd`" && cd /d `"$targetDir`" && .\scaffold-database `"$connString`""

    Write-Host "Launching VS 2022 Environment..." -ForegroundColor Cyan
    Start-Process "cmd.exe" -ArgumentList "/k `"$innerCommand`"" -NoNewWindow
    exit
}

function Invoke-GitMigrationBatch {
    <#
    .SYNOPSIS Batch moves files using mvkh.ps1, preserving git history.
    .PARAMETER SearchPattern The wildcard pattern (e.g., "*.cs").
    .PARAMETER DestinationFolder The target directory.
    .PARAMETER DryRun If specified, only lists files to be moved.
    #>
    [CmdletBinding(SupportsShouldProcess=$true)]
    param(
        [Parameter(Mandatory=$true, Position=0)] [string]$SearchPattern,
        [Parameter(Mandatory=$true, Position=1)] [string]$DestinationFolder,
        [switch]$DryRun
    )
    # ... (rest of your logic remains same)
    $filesToMove = Get-ChildItem -Path . -Filter $SearchPattern -File
    if ($DryRun) {
        $filesToMove | Select-Object Name, FullName | Format-Table -AutoSize
        return
    }
    foreach ($file in $filesToMove) {
        if ($PSCmdlet.ShouldProcess($file.Name)) {
            mvkh.ps1 -SourceFilePath $file.FullName -DestinationFile $DestinationFolder
        }
    }
}

function Get-InvoicePeriod {
    <# .SYNOPSIS Queries TRAYINVOICE for periods in a specific year (2-digit input). #>
    param(
        [Parameter(Mandatory=$true)] [ValidateRange(0, 49)] [int]$Year
    )
    $FullYear = $Year + 2000
    $connString = "Data Source=localhost;Initial Catalog=TRAYINVOICE;Integrated Security=True;TrustServerCertificate=True;Encrypt=True"
    $query = "SELECT * FROM InvoicePeriods WHERE DATEPART(year, PeriodStartDate) = $FullYear"
    Invoke-Sqlcmd -ConnectionString $connString -Query $query | Format-Table -AutoSize
}
